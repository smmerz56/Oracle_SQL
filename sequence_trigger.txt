CREATE SEQUENCE CELL_DATA.KAFKA_ANTENNA_SEQUENCE
  START WITH 1
  increment by 1
  MAXVALUE 9999999999999999999999999999
  MINVALUE 1
  NOCYCLE;


CREATE OR REPLACE TRIGGER cell_data.kafka_antenna_trigger
  BEFORE INSERT ON kafka_cell_antenna_details
  FOR EACH ROW
BEGIN
  SELECT KAFKA_CELL_ANTENNA_SEQUENCE.nextval
  INTO :new.id
  FROM dual;
END;
